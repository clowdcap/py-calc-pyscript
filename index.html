input_text = Element("input_text") op = Element("output") op.write('Selecione a botao para ver o resultado') def clear(*args, **kwargs): if input_text.value == '': op.write('') else: input_text.clear() op.write('Texto apagado') def count(*args, **kwargs): number = len(input_text.value) if number <= 0: op.write('Erro, digite seu nome') elif number < 10: op.write(f'Nome pequeno - {number} caracteres') elif number >= 10: op.write(f'Nome grande - {number} caracteres') def upper(*args, **kwargs): if input_text.value == '': op.write('Erro, digite seu nome') else: upper_text = str(input_text.value).upper() op.write(upper_text) def lower(*args, **kwargs): if input_text.value == '': op.write('Erro, digite seu nome') else: lower_text = str(input_text.value).lower() op.write(lower_text)
